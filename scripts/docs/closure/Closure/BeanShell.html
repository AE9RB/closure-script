Overview

def run(command)
  begin
    return execute command
  rescue Errno::EPIPE
    # Shut down broken pipe; another will be started.
    $cmdin.close
    $cmdout.close
    $cmderr.close
    $cmdin = nil
  end
  # This "second chance" will not rescue the error.
  execute command
end
